class PermissionUtility:
    """Utility class to check user permissions based on permission set lines"""
    
    @staticmethod
    def get_user_permission_sets(user):
        """Get all permission sets for a user"""
        permission_sets = set()
        
        user_groups = UserGroup.objects.filter(group__in=user.groups.all())
        
        for user_group in user_groups:
            permission_sets.update(user_group.permission_sets.all())
        
        return permission_sets
    
    @staticmethod
    def check_object_permission(user, object_type_code, object_id, permission_type):
        """Check if user has specific permission for an object"""
        permission_sets = PermissionUtility.get_user_permission_sets(user)
        
        # Look for permission set lines that match the object
        permission_lines = PermissionSetLine.objects.filter(
            permission_set__in=permission_sets,
            object_type__code=object_type_code,
            object_id=object_id
        )
        
        for line in permission_lines:
            permission_value = getattr(line, f"{permission_type}_permission")
            if permission_value in ['yes', 'indirect']:
                return True
        
        return False
    
    @staticmethod
    def get_user_permissions(user):
        """Get all permissions for a user in a structured format"""
        permission_sets = PermissionUtility.get_user_permission_sets(user)
        
        permissions = {}
        permission_lines = PermissionSetLine.objects.filter(
            permission_set__in=permission_sets
        ).select_related('object_type')
        
        for line in permission_lines:
            key = f"{line.object_type.code}_{line.object_id}"
            permissions[key] = {
                'object_type': line.object_type.code,
                'object_id': line.object_id,
                'object_name': line.object_name,
                'read': line.read_permission,
                'insert': line.insert_permission,
                'modify': line.modify_permission,
                'delete': line.delete_permission,
                'execute': line.execute_permission,
            }
        
        return permissions

export const PermissionProvider = ({ children }) => {
  const [permissions, setPermissions] = useState({});
  const [loading, setLoading] = useState(true);
  const [hasLoaded, setHasLoaded] = useState(false);

  const loadPermissions = async () => {
    try {
      setLoading(true);
      const response = await axios.get('/api/user-permissions/');
      setPermissions(response.data);
      setHasLoaded(true);
    } catch (error) {
      console.error('Error loading permissions:', error);
      setPermissions({});
    } finally {
      setLoading(false);
    }
  };

  const checkPermission = (objectType, objectId, permissionType) => {
    if (!hasLoaded) return false;
    
    const key = `${objectType}_${objectId}`;
    const objectPermissions = permissions[key];
    
    if (!objectPermissions) return false;

    const permissionValue = objectPermissions[permissionType];
    
    if (permissionType === 'execute') {
      return permissionValue === 'yes';
    }
    
    return ['yes', 'indirect'].includes(permissionValue);
  };

  const getObjectPermissions = (objectType, objectId) => {
    const key = `${objectType}_${objectId}`;
    return permissions[key] || null;
  };

  useEffect(() => {
    loadPermissions();
  }, []);

  const value = {
    permissions,
    loading,
    hasLoaded,
    checkPermission,
    getObjectPermissions,
    reloadPermissions: loadPermissions
  };

  return (
    <PermissionContext.Provider value={value}>
      {children}
    </PermissionContext.Provider>
  );
};
