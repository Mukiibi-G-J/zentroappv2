class BOMLine(models.Model):
    LINE_TYPES = (
        ('resource', 'Resource'),
        ('inventory', 'Inventory'),
        ('material', 'Material'),
    )
    
    bom = models.ForeignKey(ProductionBOM, on_delete=models.CASCADE, related_name='lines')
    line_type = models.CharField(max_length=10, choices=LINE_TYPES)
    
    # Resource link (if line_type == 'resource')
    resource = models.ForeignKey(Resource, on_delete=models.CASCADE, null=True, blank=True)
    resource_quantity = models.DecimalField(max_digits=8, decimal_places=3, default=0)  # 0.75 hours
    
    # Inventory link (if line_type == 'inventory')
    inventory_item = models.ForeignKey('your_app.InventoryItem', on_delete=models.CASCADE, null=True, blank=True)
    inventory_quantity = models.DecimalField(max_digits=8, decimal_places=3, default=0)  # 1.0 units
    
    unit_cost = models.DecimalField(max_digits=10, decimal_places=2, default=0)
    
    def save(self, *args, **kwargs):
        # Auto-calculate unit cost based on type
        if self.line_type == 'resource' and self.resource:
            self.unit_cost = self.resource.cost_rate
        elif self.line_type == 'inventory' and self.inventory_item:
            self.unit_cost = self.inventory_item.unit_cost
        super().save(*args, **kwargs)
    
    def __str__(self):
        return f"{self.bom.bom_code} - {self.line_type}"

Key Business Logic Methods
1. Calculate Service Cost
python
# models.py - in ProductionBOM class
def calculate_total_cost(self):
    total_cost = 0
    for line in self.lines.all():
        if line.line_type == 'resource':
            total_cost += line.resource_quantity * line.resource.cost_rate
        elif line.line_type == 'inventory':
            total_cost += line.inventory_quantity * line.inventory_item.unit_cost
    return total_cost

def calculate_profit_margin(self):
    service_price = self.service_item.base_price
    total_cost = self.calculate_total_cost()
    if service_price > 0:
        return ((service_price - total_cost) / service_price) * 100
    return 0
2. Update Inventory on Service Sale

# In your POS sale processing logic
def process_service_sale(service_item, quantity=1):
    """
    Called when a service is sold to update inventory and track resource usage
    """
    if not service_item.uses_production_bom:
        return
    
    bom = service_item.production_bom
    for line in bom.lines.all():
        if line.line_type == 'inventory' and line.inventory_item:
            # Deduct inventory used in the service
            line.inventory_item.quantity_on_hand -= (line.inventory_quantity * quantity)
            line.inventory_item.save()
        
        # You could also track resource utilization here
        if line.line_type == 'resource':
            log_resource_usage(line.resource, line.resource_quantity * quantity)

    unit_price = models.DecimalField(max_digits=10, decimal_places=2)
    
    @property
    def total_price(self):
        return self.quantity * self.unit_price
    
    def __str__(self):
        if self.line_type == 'product':
            return f"{self.product.name} x {self.quantity}"
        else:
            return f"{self.service.name} by {self.resource.name if self.resource else 'TBD'}"
